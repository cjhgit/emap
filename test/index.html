$('#china-map-size').emap({
    width: 800,
    height: 800
});

显示名称


$('#china-map-name').emap({
    showName: true,
});

$('#china-map-hover').emap({
    defaultColor: '#67eb6a',
    hoverColor: '#24d228'
});

$('#china-map-colorful').emap({
    colorful: true
});

$('#china-map-color').emap({
    data: {
        guangdong: {
            color: '#f34951'
        },
        beijing: {
            color: '#497df3'
        }
    }
});

    	alert(province.name);
